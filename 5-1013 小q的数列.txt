#include<bits/stdc++.h>
using namespace std;
long f(long n)
{
    if (n<=1)
        return n;
    else if (n%2==0)
        return f(n/2);
    else
        return f(n/2)+1;
}
int main()
{
    long t,sum=0;
    long n,m;
    cin>>t;
    while (t--)
    {
        cin>>n;
        m=f(n);
        cout<<m<<" ";
        while (n)
        {
            if  (n%2==1)
                sum++;
            n/=2;
        }
        cout<<long(pow(2,sum))-1<<endl;//pow为double型
        sum=0;
    }
    return 0;
}
